1 /**
  2  * @fileoverview LMS SCORM Javascript API
  3  * Facilitates communication between SCORM content and an LMS.
  4  *
  5  * @author nsweany
  6  * @version 1.0.0
  7  */
  8 
  9 /**
 10  * Facilitates communication between SCORM content and an LMS.
 11  * @constructor
 12  * @param {Object} scormBehavior	a Scorm Behavior object that provides the implementation code for communication using a specific version of SCORM
 13  * @return {LMS | Object}
 14  * @example
 15  * // Define the sco object
 16  * var sco = null;
 17  * // run this function when the page loads
 18  * function scoInit(){
 19  *     try {
 20  *         sco = new LMS(new Scorm12());
 21  *         // now initialize the session (optional)
 22  *         sco.initialize();
 23  *     } catch(e){
 24  *         alert(e.toString());
 25  *     }
 26  * }
 27  */
 28 function LMS(scormBehavior){
 29 	this._scormBehavior = scormBehavior;
 30 }
 31 
 32 /**
 33  * Initializes the SCORM session with the LMS
 34  */
 35 LMS.prototype.initialize = function(){
 36 	return this._scormBehavior.initialize();
 37 }
 38 
 39 /**
 40  * Commits persistent data to the LMS
 41  */
 42 LMS.prototype.commit = function(){
 43 	return this._scormBehavior.commit();
 44 }
 45 
 46 /**
 47  * Finishes the SCORM session with the LMS
 48  */
 49 LMS.prototype.finish = function(){
 50 	return this._scormBehavior.finish();
 51 }
 52 
 53 /**
 54  * Getter method for retrieving cmi data from the LMS
 55  *
 56  * @param {String} prop	name of the cmi data property to retrieve
 57  */
 58 LMS.prototype.get = function(prop){
 59 	return this._scormBehavior.get(prop);
 60 }
 61 
 62 /**
 63  * Setter method for sending cmi data to the LMS
 64  *
 65  * @param {String} prop	name of the cmi data property to set
 66  * @param {String || Number} val	value of the cmi data property
 67  */
 68 LMS.prototype.set = function(prop, val){
 69 	return this._scormBehavior.set(prop, val);
 70 }
 71 
 72 /**
 73  * Retrieves the error code for the last SCORM call made to the LMS
 74  */
 75 LMS.prototype.errorCode = function(){
 76 	return this._scormBehavior.errorCode();
 77 }
 78 
 79 /**
 80  * Searches and returns the location of the SCORM API within the frameset heirarchy
 81  *
 82  * @param {Object} win	HTML window object to use as the starting point for the search
 83  * @return	{Object}	Reference to the SCORM API
 84  */
 85 LMS.prototype.findAPI = function(win){
 86 	return this._scormBehavior.findAPI(win);
 87 }
 88 
 89 /**
 90  * @private
 91  * Internal LMS instance of the SCORM Behavior.
 92  * Provides the actual implementation for each method
 93  */
 94 LMS.prototype._scormBehavior = null;
